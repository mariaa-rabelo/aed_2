551 * all outgoing and incoming edges.
552 * Returns true if successful, and false if such vertex does not exist.

554template <class T>
555bool Graph<T>::removeVertex(const T &in) {
556 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++)
557 if ((*it)->info == in) {
558 auto v = *it;
559 vertexSet.erase(it);
560 for (auto u : vertexSet)
561 u->removeEdgeTo(v);
562 delete v;
563 return true;
564 }
565 return false;

568template <class T>
569std::vector<Edge<Airport>> Graph<T>::findShortestPath(Vertex<Airport>* src, Vertex<Airport>* dest) {
570 //para cada vertice, seu predecessor no caminho mais curto
571 std::map<Vertex<Airport>*, Vertex<Airport>*> predecessors;
572
573 // entre src ate cada outro airport
574 std::map<Vertex<Airport>*, int> distances;
575 std::queue<Vertex<Airport>*> queue;
576
577 for (auto& vertex : vertexSet) {
578 distances[vertex] = std::numeric_limits<int>::max();
579 vertex->setVisited(false);
580 }
581
582 distances[src] = 0;
583 queue.push(src);
584 src->setVisited(true);
585
586 while (!queue.empty()) {
587 Vertex<Airport>* current = queue.front();
588 queue.pop();
589
590 for (const Edge<Airport>& edge : current->getAdj()) {
591 Vertex<Airport>* neighbor = edge.getDest();
592 if (!neighbor->isVisited()) {
593 neighbor->setVisited(true);
594 queue.push(neighbor);
595 distances[neighbor] = distances[current] + 1;
596 predecessors[neighbor] = current;
597 }
598 }
599 }
600
601 std::vector<Edge<Airport>> path;
602 if (distances[dest] != std::numeric_limits<int>::max()) {
603 Vertex<Airport>* current = dest;
604 while (current != src) {
605 Vertex<Airport>* prev = predecessors[current];
606 for (const Edge<Airport>& edge : prev->getAdj()) {
607 if (edge.getDest() == current) {
608 path.push_back(edge);
609 break;
610 }
611 }
612 current = prev;
613 }
614 std::reverse(path.begin(), path.end());
615 } else {
616 std::cout << "No path found between this source and destination :(" << std::endl;
617 }
618 return path;
619}

621template <class T>
622void Graph<T>::findAndDisplayBestPaths(const std::set<Vertex<Airport>*>& srcVertices, const std::set<Vertex<Airport>*>& destVertices) {
623 std::vector<std::vector<Edge<Airport>>> bestPaths;
624 int minStops = std::numeric_limits<int>::max();
625
626 for (auto srcVertex : srcVertices) {
627 for (auto destVertex : destVertices) {
628 std::vector<Edge<Airport>> path = findShortestPath(srcVertex, destVertex);
629 int stops = path.size() - 1;
630
631 if (stops < minStops) {
632 minStops = stops;
633 bestPaths.clear();
634 bestPaths.push_back(path);
635 } else if (stops == minStops) {
636 bestPaths.push_back(path);
637 }
638 }
639 }
640
641 for (const auto& path : bestPaths) {
642 std::cout << "Path with " << minStops << " stops: " << std::endl;
643 for (const auto& edge : path) {
644 std::cout << "Flight to " << edge.getDest()->getInfo().getCode()
645 << " (" << edge.getDest()->getInfo().getCity() << ") "
646 << " with airline " << edge.getAirline()->getCode()
647 << " (" << edge.getAirline()->getCallsign() << ")"
648 << std::endl;
649 }
650 std::cout << std::endl;
651 }
