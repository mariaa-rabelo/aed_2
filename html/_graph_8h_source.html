530template <class T>

532 for (auto it = adj.begin(); it != adj.end(); it++)

570/*
571 * Removes a vertex with a given content (in) from a graph (this), and
572 * all outgoing and incoming edges.
573 * Returns true if successful, and false if such vertex does not exist.
574 */
575template <class T>
576bool Graph<T>::removeVertex(const T &in) {
577 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++)
578 if ((*it)->info == in) {
579 auto v = *it;
580 vertexSet.erase(it);
581 for (auto u : vertexSet)
582 u->removeEdgeTo(v);
583 delete v;
584 return true;
585 }
586 return false;
587}

589/*
590 * Auxiliary function that visits a vertex (v) and its adjacent, recursively.
591 * Updates a parameter with the list of visited node contents.
592 */
593template <class T>
594void Graph<T>::dfsVisit(Vertex<T> *v, std::vector<T> & res) const {
595 v->visited = true;
596 res.push_back(v->info);
597 for (auto & e : v->adj) {
598 auto w = e.dest;
599 if ( ! w->visited)
600 dfsVisit(w, res);
601 }
602}
